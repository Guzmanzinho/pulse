READ / SELECT
============================================================

-- 1. Listar todos os utilizadores ativos.
SELECT utilizador_id, nome_utilizador, email, nome, criado_em
FROM utilizador
WHERE ativo = TRUE
ORDER BY criado_em DESC;

-- 2. Consultar dados de login por nome de utilizador.
SELECT utilizador_id, nome_utilizador, password_hash, ativo
FROM utilizador
WHERE nome_utilizador = 'ana_silva';

-- 3. Listar tweets ativos com o nome do autor.
SELECT t.tweet_id, u.nome_utilizador, t.conteudo, t.criado_em
FROM tweet t
INNER JOIN utilizador u ON u.utilizador_id = t.utilizador_id
WHERE t.ativo = TRUE
ORDER BY t.criado_em DESC;

-- 4. Mostrar feed de um utilizador com tweets de quem ele segue.
SELECT t.tweet_id, u.nome_utilizador, t.conteudo, t.criado_em
FROM seguimento s
INNER JOIN tweet t ON t.utilizador_id = s.seguido_id
INNER JOIN utilizador u ON u.utilizador_id = t.utilizador_id
WHERE s.seguidor_id = 2
  AND t.ativo = TRUE
  AND u.ativo = TRUE
ORDER BY t.criado_em DESC;

-- 5. Contar gostos por tweet.
SELECT t.tweet_id, t.conteudo, COUNT(g.utilizador_id) AS total_gostos
FROM tweet t
LEFT JOIN gosto g ON g.tweet_id = t.tweet_id
GROUP BY t.tweet_id, t.conteudo
ORDER BY total_gostos DESC;

-- 6. Listar seguidores de um utilizador.
SELECT u.utilizador_id, u.nome_utilizador, u.nome, s.criado_em AS segue_desde
FROM seguimento s
INNER JOIN utilizador u ON u.utilizador_id = s.seguidor_id
WHERE s.seguido_id = 2
ORDER BY s.criado_em DESC;

-- 7. Listar tweets com imagem.
SELECT t.tweet_id, u.nome_utilizador, t.conteudo, i.url_imagem, i.texto_alt
FROM tweet t
INNER JOIN utilizador u ON u.utilizador_id = t.utilizador_id
INNER JOIN imagem_tweet i ON i.tweet_id = t.tweet_id
WHERE t.ativo = TRUE
ORDER BY t.criado_em DESC;

============================================================

  AND tweet_id = 3;

  AND seguido_id = 4;
